fun print_it<T>(x: T) void {
    x.print();
}

struct Person {
    name: str,
    age: u8
}

let person: Person = {
    name = "John"
    age = 25
}

let number: i32 = 42;

print_it(person);  // Should work - Person can have print method
print_it(number);  // Should error - i32 doesn't have print method
