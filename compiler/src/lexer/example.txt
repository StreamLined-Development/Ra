generic F {f16, f32, f64, f128}
generic I {i8, i16, i32, i64, i128}
generic U {u8, u16, u32, u64, u128}

fum sign::<F>(x: F) str {if x > 0.0 return "positive"};
fum sign::<F>(x: F) str {if x < 0.0 return "negative"};
fum sign::<I>(x: I) str {if x > 0 return "positive"};
fum sign::<I>(x: I) str {if x < 0 return "negative"};
fum sign::<F>(0.0: F) str {return "zero"};
fum sign::<I>(0: I) str {return "zero"};
